{

    while(~scanf("%lf %lf %lf %lf",&x1,&x2,&y1,&y2))

    }

}

2007-平方和与立方和

#include<stdio.h>
int main()
{
    int m,n;
    while(~scanf("%d%d",&m,&n))
    {
        int i,j=0,k=0;
        if(m<=n)
        for(i=m;i<=n;i++)
        {
            if(i%2==0)
                j=j+i*i;
            else if(i%2!=0)
                k=k+i*i*i;    
        }
        else 
        for(i=n;i<=m;i++)
        {
            if(i%2==0)
                j=j+i*i;
            else if(i%2!=0)
                k=k+i*i*i;    
        }
        printf("%d %d\n",j,k);
    }
    return 0;
}

2008-数值统计

#include<stdio.h>
int main()
{
    int n;
    double a[105];

    while(~scanf("%d",&n))
    {
        int b=0,c=0,d=0;
        int i=0;
        for(i;i<n;i++)
        {
            scanf("%lf",&a[i]);
            if(a[i]<0)
                b++;
            else if(a[i]==0)
                c++;
            else d++;
        }
        printf("%d %d %d\n",b,c,d);
    }
    return 0;
}
